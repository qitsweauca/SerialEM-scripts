ScriptName LD-Group

# macro to skip points except the very first in the group.
# assume LD is setup.

# define ref buffer 
buffer = T

#### X,Y position 
RealignToNavItem 
ResetImageShift
Copy A $buffer
CallFunction MyFuncs::Relax
CallFunction MyFuncs::BufferShot $buffer
AlignTo $buffer 0 1

# preparation for first item in group
ReportGroupStatus 
If $repVal1 == 1 OR $repVal1 == 0
    #Call Z_byV
    #UpdateGroup Z
    AutoCenterBeam
    CallFunction MyFuncs::CycleTargetDefocus -1.2 -2.0 0.2
    G
Else 
    echo Directly shot!
Endif

# Single shot For K2, uncomment next line
EarlyReturnNextShot 0
R

# multi-shot
